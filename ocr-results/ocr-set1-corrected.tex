\section*{图片 2（教材第十一章习题）}

\fbox{%
\parbox{\dimexpr\textwidth-2\fboxsep-2\fboxrule\relax}{%
\textbf{第3题.} 计算
\[
\oint_L (2xy - x^2)\,dx + (x + y^2)\,dy
\]
$L$ 是由抛物线 $y^2 = x$ 和 $y = x^2$ 所围成区域的边界曲线，
方向为\textbf{逆时针}。

\medskip

\textbf{第4题.} 计算
\[
\int_L (2a - y)\,dx + x\,dy
\]
其中 $L$ 为摆线 $x = a(t - \sin t)$，$y = a(1 - \cos t)$
上由 $t = 0$ 到 $t = 2\pi$ 的一段弧。

\medskip

\textbf{第5题.} 计算
\[
\int_L y\,dx + z\,dy + x\,dz
\]
其中 $L$ 为螺旋线 $x = a\cos t$，$y = a\sin t$，$z = bt$
从 $t = 0$ 到 $t = 2\pi$ 的弧段。

\medskip

\textbf{第6题.} 计算
\[
\int_L (y^2 - z^2)\,dx + 2yz\,dy - x^2\,dz
\]
其中 $L$ 为曲线 $\begin{cases} y = x^2 \\ z = x^3 \end{cases}$
从点 $(0,0,0)$ 到点 $(1,1,1)$ 的弧段。%
}%
}

% ==================== 图片 3 ====================
\newpage
\section*{图片 3（高等数学练习与提高 三）}

\fbox{%
\parbox{\dimexpr\textwidth-2\fboxsep-2\fboxrule\relax}{%
\textbf{第1题.}
力场由方向为 $y$ 轴的负方向、大小等于作用点横坐标的平方构成。
设质点质量为 $m$，沿抛物线 $1 - x = y^2$ 从点 $(1,0)$ 移动到点 $(0,1)$，
求力场所做的功。

\medskip

\textbf{第2题.}
将曲线积分
\[
I = \int_L \bigl[1 + (xy + y^2)\sin x\bigr]\,dx
      + \bigl[(x^2 + xy)\sin y\bigr]\,dy
\]
化为形如 $\displaystyle\int_L f(x)\,dx + g(y)\,dy$ 的积分，并计算它。
其中 $L$ 为上半椭圆 $x^2 + xy + y^2 = 1$，
从点 $(-1,0)$ 到点 $(1,0)$ 的弧段。

\bigskip

\textbf{注意：} 化成的积分形式为 $\int_L f(x)\,dx + g(y)\,dy$，
含 $dx$ 的项仅为 $x$ 的函数 $f(x)$，
含 $dy$ 的项仅为 $y$ 的函数 $g(y)$。

\medskip

\textbf{第3题.}
将下列对坐标的曲线积分化为对弧长的曲线积分：

\begin{enumerate}[label=(\arabic*),leftmargin=*]
  \item 计算 $\displaystyle\int_L 3x^2y\,dx + y^2\,dy$，
        $L$ 为曲线 $y^2 = x$ 从点 $(0,0)$ 到点 $(1,1)$ 的弧段。
  \item 计算 $\displaystyle\int_L P\,dx + Q\,dy + R\,dz$，
        $L$ 为曲线 $x = t$，$y = t^2$，$z = t^3$，
        $t$ 从 $0$ 到 $1$ 的弧段。
\end{enumerate}%
}%
}

\section*{全部题目汇总}

\begin{enumerate}[leftmargin=*,itemsep=6pt]
  \item[\textbf{图1}] $\displaystyle\int_L x\sin y\,dx - y\sin x\,dy$，
        $L$：$A(\pi,0) \to B(0,\pi)$ 直线段
  \item[\textbf{图2-3}] $\displaystyle\oint_L (2xy - x^2)dx + (x + y^2)dy$，
        $L$：$y^2=x$ 与 $y=x^2$ 围成边界，\textbf{逆时针}
  \item[\textbf{图2-4}] $\displaystyle\int_L (2a - y)dx + x\,dy$，
        $L$：摆线 $x=a(t-\sin t), y=a(1-\cos t)$，$t:0\to2\pi$
  \item[\textbf{图2-5}] $\displaystyle\int_L y\,dx + z\,dy + x\,dz$，
        $L$：螺旋线 $x=a\cos t, y=a\sin t, z=bt$，$t:0\to2\pi$
  \item[\textbf{图2-6}] $\displaystyle\int_L (y^2-z^2)dx + 2yz\,dy - x^2\,dz$，
        $L$：$y=x^2, z=x^3$，$(0,0,0)\to(1,1,1)$
  \item[\textbf{图3-1}] 变力做功：$\vec{F}=(0,-x^2)$，质量 $m$，
        路径 $1-x=y^2$，$(1,0)\to(0,1)$
  \item[\textbf{图3-2}] 将 $\int_L [1+(xy+y^2)\sin x]dx + [(x^2+xy)\sin y]dy$
        化为 $\int_L f(x)dx+g(y)dy$，
        $L$：上半椭圆 $x^2+xy+y^2=1$，$(-1,0)\to(1,0)$
  \item[\textbf{图3-3}] 将对坐标的曲线积分化为对弧长的曲线积分（2小题）
\end{enumerate}
